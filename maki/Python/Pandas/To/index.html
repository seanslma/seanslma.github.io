    #each col to a key
df = pd.DataFrame({'a': [1,2],'b': ['x', 'y'], 'c': ['u', 'v']})
dic = df.to_dict('list')

#each row to a key
df = pd.DataFrame({'id': ['p', 'q', 'r'],'a': ['red', 'yellow', 'blue'], 'b': [0.5, 0.25, 0.125]})
dic = df.set_index('id').T.to_dict('list')

#not work
